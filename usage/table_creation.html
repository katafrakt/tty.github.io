To instantiate table pass 2-dimensional array:

{% highlight ruby %} table = TTY::Table[['a1', 'a2'], ['b1', 'b2']] table = TTY::Table.new [['a1', 'a2'], ['b1', 'b2']] table = TTY::Table.new rows: [['a1', 'a2'], ['b1', 'b2']] table = TTY::Table.new ['h1', 'h2'], [['a1', 'a2'], ['b1', 'b2']] table = TTY::Table.new header: ['h1', 'h2'], rows: [['a1', 'a2'], ['b1', 'b2']] {% endhighlight %}
